--- Check execution plans
SELECT 
    deqs.query_hash, 
    deqs.query_plan_hash, 
    deqp.query_plan, 
    dest.text 
FROM 
    sys.dm_exec_query_stats AS deqs 
CROSS APPLY 
    sys.dm_exec_query_plan(deqs.plan_handle) AS deqp 
CROSS APPLY 
    sys.dm_exec_sql_text(deqs.sql_handle) AS dest;
WHERE text = '$$'
